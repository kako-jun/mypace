function importSettings(json: string): boolean {
  try {
    const data = JSON.parse(json)
    if (!data.mypace_settings) return false

    const settings = data.mypace_settings

    if (settings.theme) {
      applyThemeColors(settings.theme.colors)
      setThemeMode(settings.theme.mode)
    }

    if (settings.display) {
      applyDisplaySettings(settings.display)
    }

    if (settings.filters) {
      applyFilterSettings(settings.filters)
    }

    return true
  } catch {
    return false
